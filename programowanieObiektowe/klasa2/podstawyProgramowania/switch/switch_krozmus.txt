#include <cmath>
#include <iomanip>
using namespace std;

int main() {
    float a;
    float b;
    string oper;

    cout << "Podaj pierwszą liczbę działania: \n";
    cin >> a;
    cout << "Podaj drugą liczbę działania: \n";
    cin >> b;
    cout << "Wybierz operację: \n";
    cout << "- + * / \n";
    
    cin >> oper;

    if(oper == "-") {
        cout << "Wynik działania to: " << a-b << endl;
    }
    if(oper == "+") {
        cout << "Wynik działania to: " << a+b << endl;
    }
    if(oper == "*") {
        cout << "Wynik działania to: " << a*b << endl;
    }
    if(oper == "/") {
        cout << "Wynik działania to: " << a/b << endl;
    }
    
    return 0;
}

#include <iostream>
#include <cmath>

using namespace std;

int main() {
    int a;
    int b;
    int h;
    
    char polec;
    
    cout << "Wybierz jakiej figury chciałbyś obliczyć pole\n" ;
    cout << "Figury do wyboru: \na) Kwadrat\nb) Prostokąt\nc) Trójkąt\nd) Trapez\n";
    cout << "Wybrana figura: ";
    cin >> polec;
    
    switch(polec)
    {
        case 'a':
            cout << "Podaj bok kwadratu: ";
            cin >> a;
            cout << "\nPole tego kwadratu to: " << a*a << endl;
            break;
        case 'b':
            cout << "Podaj pierwszy bok prostokąta: ";
            cin >> a;
            cout << "Podaj drugi bok prostokąta: ";
            cin >> b ;
            cout << "\nPole tego prostokąta to: " << a*b << endl;
            break;
        case 'c':
            cout << "Podaj podstawę trójkąta: ";
            cin >> a;
            cout << "Podaj wysokość trójkąta: ";
            cin >> h;
            cout << "\nPole tego trójkąta to: " << (a*h)/2 << endl;
            break;
        case 'd':
            cout << "Podaj podstawę trapeza: ";
            cin >> a;
            cout << "Podaj drugi bok trapeza: ";
            cin >> b;
            cout << "Podaj wysokość trapezu: ";
            cin >> h;
            cout << "\nPole tego trapeza to: " << ((a+b)*h)/2 << endl;
            break;
        default:
            cout<<"Wybrałeś niepoprawną figurę";
    }
    return 0;
}
