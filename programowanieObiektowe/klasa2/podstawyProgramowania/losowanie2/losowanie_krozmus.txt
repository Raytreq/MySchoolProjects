#include <iostream>
#include <cstdlib>
#include <ctime>

using namespace std;

bool pier(int x) {
    if(x<2) {
        return false;
    };

    for(int i=2;i*i<=x;i++) {
        if(x%i==0) {
            return false;
        };
    };
    return true;
};

int main() {
    srand(time(NULL));
    int n = rand() % 999+1;
    
    cout <<"Wylosowana liczba wynosi: "<< n<< endl;

    if(pier(n)==1)
        cout<< "Wylosowana liczba ("<< n <<") jest liczbą pierwszą."<<endl;
    else
        cout<< "Wylosowana liczba ("<< n <<") nie jest liczbą pierwszą."<<endl;
