#include <iostream>
using namespace std;

int main() {
    int x;
    int n;
    int ilosc = 0;
    int najwieksza = 0;
    cout<<"Podaj ilość liczb które podasz: "<<endl;
    cin>>n;
    int liczby[n];

    cout<<"Podaj liczby całkowite: "<<flush;
    for(int i = 0;i < n;i++){
        cin>>x;
        liczby[i] = x;
    }
    cout<<"Podałeś: "<<endl;
    for(int i = 0;i < n;i++){
        cout<<liczby[i]<<" "<<flush;
    }

    for(int i=0; i<n; i++){
        if (najwieksza < liczby[i]) {
            najwieksza = liczby[i];
        }
    }

    for(int i=0; i<n; i++){
        if (liczby[i] == najwieksza){
            ilosc+=1;
        }
    }

    cout<<endl<<"Największa podana liczba to "<<najwieksza<<", która wystąpiła "<<ilosc<<" razy"<<endl;

    return 0;
};
