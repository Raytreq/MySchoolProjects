import string

zdanie = input("Podaj dowolne zdanie: ")
alfabet_string_lower = string.ascii_lowercase
alfabet_list_lower = list(alfabet_string_lower)
alfabet_string_upper = string.ascii_uppercase
alfabet_list_upper = list(alfabet_string_upper)

for i in zdanie:
    if i in alfabet_list_lower:
        alfabet_list_lower.remove(i)

for i in zdanie:
    if i in alfabet_list_upper:
        alfabet_list_upper.remove(i)

print("Małe litery alfabetu, których nie użyto w podanym zdaniu: \n",alfabet_list_lower,"\n Duże litery alfabetu, których nie użyłeś w swoim zdaniu: \n ", alfabet_list_upper)
